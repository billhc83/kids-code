// --- CODEBREAKER SYSTEM ---

String answer = "SPARK";
int likeness = 0;
bool solved = false;

void setup() {
  Serial.begin(9600);

  Serial.println("================================");
  Serial.println("     C O D E  B R E A K E R    ");
  Serial.println("================================");
  Serial.println("Find the hidden 5-letter word.");
  Serial.println("");

  Serial.println("X K Q S P A R K M Z");
  Serial.println("B R T F L A M E Q X");
  Serial.println("P Q S P A R K T Z R");
  Serial.println("W Z X B R A N D T P");
  Serial.println("S P A R K X Q Z B M");
  Serial.println("T R X N G L O W K B");
  Serial.println("Q Z B S P A R K X T");
  Serial.println("M X T R B L A Z E P");
  Serial.println("B T Z X Q M R N V K");
  Serial.println("P N V Q Z B X T M R");

  Serial.println("");
  Serial.println("Enter your guess:");
}

void loop() {
  if (Serial.available() > 0) {

    String guess = Serial.readString();
    guess.trim();
    guess.toUpperCase();

    likeness = 0;

    for (int i = 0; i < 5; i++) {
      if (guess[i] == answer[i]) {
        likeness++;
      }
    }

    Serial.print("Likeness = ");
    Serial.println(likeness);

    if (likeness == 5) {
      Serial.println("CODE CRACKED! ACCESS GRANTED.");
      solved = true;
    } else {
      Serial.println("Try again:");
    }
  }
}

🛠️ Upgrade Your Cipher


Right now, your system always uses the same secret word and grid. That means once someone solves it… they already know the answer.

Real agents change their systems to stay ahead.


🔄 Change the Secret Code

  • Pick a new 5-letter word
  • Make sure it matches the format (all letters)
  • Update the hidden answer in your system

🧩 Change the Grid

  • Hide your new word inside the letter grid
  • Mix in random letters to confuse the player
  • Make sure your word appears clearly at least once

🎯 Make It Harder

  • Use tricky letters (Z, X, Q)
  • Add multiple fake patterns
  • When you change the cipher make sure you use CAPITAL letters
